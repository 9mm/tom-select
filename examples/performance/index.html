Large Dataset Performance

This shows how it performs with 25,000 items.

  • JavaScript
  • HTML 
var letters = 'abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUV';
var options = [];
for (var i = 0; i < 25000; i++) {
	var title = [];
	for (var j = 0; j < 8; j++) {
		title.push(letters.charAt(Math.round((letters.length - 1) * Math.random())));
	}
	options.push({
		id: i,
		title: title.join('')
	});
}

tomSelect('#select-junk',{
	maxItems: null,
	maxOptions: 100,
	valueField: 'id',
	labelField: 'title',
	searchField: 'title',
	sortField: 'title',
	options: options,
	create: false
});
